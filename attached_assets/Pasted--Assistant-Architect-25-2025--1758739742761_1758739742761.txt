export function calculateDistance(p1: [number, number], p2: [number, number]): number {
  return Math.sqrt(Math.pow(p2[0] - p1[0], 2) + Math.pow(p2[1] - p1[1], 2));
}

export function calculateAngle(p1: [number, number], p2: [number, number]): number {
  const deltaX = p2[0] - p1[0];
  const deltaY = p2[1] - p1[1];
  let angle = Math.atan2(deltaY, deltaX) * (180 / Math.PI);
  return angle >= 0 ? angle : angle + 360;
}

export function classifyAngle(angle: number): number {

}

export function calculateArea(points: [number, number][]): number {
  let area = 0;
  const n = points.length;
  for (let i = 0; i < n; i++) {
    const [x1, y1] = points[i];
    const [x2, y2] = points[(i + 1) % n];
    area += x1 * y2 - y1 * x2;
  }
  return Math.abs(area) / 2;
}
